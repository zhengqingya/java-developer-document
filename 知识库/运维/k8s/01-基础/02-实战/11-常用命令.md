# kubectl 常用命令

> 见 https://kubernetes.io/zh-cn/docs/reference/kubectl/generated/kubectl/

```shell
# 打印服务器支持的 API 资源
kubectl api-resources
```

### 命名空间

```shell
# 查看命名空间
kubectl get ns

# 查看默认命名空间`default`下的应用 -- 如果操作不指定命名空间，默认都是放在`default`命名空间下
kubectl get pods
# 查看所有命名空间下的应用
kubectl get pods -A
# 查看指定命名空间`xxx`下的应用
kubectl get pods -n xxx
# NAME                                    READY   STATUS    RESTARTS   AGE
# demo-k8s-otel-go-565df8dd79-q75mc       1/1     Running   0          2m10s
# demo-k8s-otel-java-86796c575b-575f7     1/1     Running   0          2m11s
# demo-k8s-otel-php-7f975849-8jpxf        1/1     Running   0          2m10s
# demo-k8s-otel-python-7b7f954846-hjqtk   1/1     Running   0          2m10s
# 在标准列基础上增加IP、节点等关键调度信息
kubectl get pods -n xxx -o wide
# NAME                                    READY   STATUS    RESTARTS   AGE   IP            NODE                    NOMINATED NODE   READINESS GATES
# demo-k8s-otel-go-565df8dd79-q75mc       1/1     Running   0          50s   10.244.0.37   desktop-control-plane   <none>           <none>
# demo-k8s-otel-java-86796c575b-575f7     1/1     Running   0          51s   10.244.0.35   desktop-control-plane   <none>           <none>
# demo-k8s-otel-php-7f975849-8jpxf        1/1     Running   0          50s   10.244.0.38   desktop-control-plane   <none>           <none>
# demo-k8s-otel-python-7b7f954846-hjqtk   1/1     Running   0          50s   10.244.0.36   desktop-control-plane   <none>           <none>


# 创建命名空间 `zq`
kubectl create ns zq
# 删除 -- 会将其下所有应用删除
kubectl delete ns zq


# 在K8s集群中，对指定命名空间下的 Deployment 执行“无感滚动重启”。 eg：让注入配置生效
kubectl rollout restart deployment demo-java -n zq

# 把 k8s集群里 zq命名空间中 demo-java Service 的 8080端口，映射到你本地 Mac 的 18080端口，让你能在本地直接访问这个服务。
# kubectl port-forward 端口转发。 访问 http://localhost:18080 -> 实际请求会被转发到 demo-java.zq.svc.cluster.local:8080
kubectl port-forward -n zq svc/demo-java 18080:8080 

# 查看最近 20 行日志，并实时跟踪后续新日志
kubectl logs -n zq deploy/demo-java --tail=20 -f

# 查看 zq 命名空间中，标签为 app=demo-java 的 Pod 详细信息
kubectl describe pod -n zq -l app=demo-java
# 查看指定名称的单个 Pod 详情
kubectl describe pod -n zq demo-java-xxx 

```

### 问题排查

```shell
# 1. 持续观察 Pod 的启动状态，按 Ctrl+C 退出
kubectl get pods -n zq -w

# 2. 查看 namespace 中最近发生的事件，快速定位调度、镜像、存储和健康检查问题
kubectl get events -n zq --sort-by=.lastTimestamp | tail -30

# 3. 深入查看具体 Pod 的容器状态、镜像、环境变量、挂载和事件
kubectl describe pod -n zq <pod-name>

# 4. 容器已经启动过时，查看应用内部的运行日志
kubectl logs -n zq <pod-name>
```
